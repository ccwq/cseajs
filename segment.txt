/**
 * 遍历给定document下面的iframe
 * @param doc root document
 * @param maxDeep [可选]最大遍历深度
 * @param callback 执行操作
 */
function travelDepIframe(doc,maxDeep,callback){
    if(typeof maxDeep == "function"){
        callback = maxDeep;
        maxDeep = 0;
    }

    var deep = 0;
    function travel(doc,deep){
        callback.call(doc,doc);
        if(maxDeep && deep>=maxDeep)   return;
        $("iframe",doc).each(function(){
            travel(this.contentWindow.document, deep+1);
        });
    }
    travel(document,1);
}
